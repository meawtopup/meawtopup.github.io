// ==UserScript==
// @name         Click Farm 9.0 | TDDFarm
// @namespace    http://tampermonkey.net/
// @version      9.0
// @description  TDDFarm
// @author       MobyEX
// @match        https://www.torrentdd.com/farm.php*
// @match        https://www.torrentdd.com/chat.php*
// @icon         https://www.google.com/s2/favicons?sz=64&domain=torrentdd.com
// @grant        none
// ==/UserScript==

(function () {
    'use strict';

    let timerInterval = null;
    let wakeLock = null;

    const style = document.createElement('style');
    style.innerHTML = `
        @keyframes tddBlink { 0% { opacity: 1; } 50% { opacity: 0.2; } 100% { opacity: 1; } }
        .tdd-reloading-msg {
            animation: tddBlink 0.8s infinite;
            color: #ffeb3b;
            font-weight: bold;
            white-space: nowrap;
        }
    `;
    document.head.appendChild(style);

    async function fetchFarmDoc() {
        try {
            const res = await fetch('/farm.php');
            const text = await res.text();
            return new DOMParser().parseFromString(text, 'text/html');
        } catch (e) {
            console.error('TDDFarm: Failed to fetch farm.php', e);
            return document;
        }
    }

    function getHarvestableIDs(doc) {
        return Array.from(doc.querySelectorAll('a[onclick*="action=store"]'))
            .map(el => {
                const match = el.getAttribute('onclick').match(/ground=(\d+)/);
                return match ? match[1] : null;
            }).filter(id => id !== null);
    }

    function getPlantableIDs(doc) {
        return Array.from(doc.querySelectorAll('button[onclick*="action=seed"]'))
            .map(el => {
                const match = el.getAttribute('onclick').match(/ground=(\d+)/);
                return match ? match[1] : null;
            }).filter(id => id !== null);
    }

    async function processActions(ids, actionType) {
        for (let id of ids) {
            await fetch(`/farm.php?action=${actionType}&ground=${id}`);
            console.log(`Action: ${actionType} on Plot: ${id} Done.`);
        }
    }

    function startReloadCountdown() {
        if (timerInterval) clearInterval(timerInterval);
        const container = document.getElementById('tdd-farm-bot-item-content');
        if (!container) { location.reload(); return; }

        let seconds = 10;
        const updateCountdown = () => {
            container.innerHTML = `<span class="tdd-reloading-msg">⚠️ กำลังจะรีโหลดหน้าใหม่ ใน ${seconds} ⚠️</span>`;
            if (seconds <= 0) {
                location.reload();
            } else {
                seconds--;
                setTimeout(updateCountdown, 1000);
            }
        };
        updateCountdown();
    }

    async function initBot() {
        if (timerInterval) clearInterval(timerInterval);

        const isFarmPage = window.location.pathname.includes('farm.php');
        const farmDoc = isFarmPage ? document : await fetchFarmDoc();

        const oldContainer = document.getElementById('tdd-farm-bot-item');
        if (oldContainer) oldContainer.remove();

        const container = document.createElement('li');
        container.id = 'tdd-farm-bot-item';
        container.className = 'nav-item d-flex align-items-center';

        const content = document.createElement('div');
        content.id = 'tdd-farm-bot-item-content';
        content.style = 'display: flex; align-items: center; gap: 6px; margin-right: 8px; font-size: 12px; color: #fff;';
        container.appendChild(content);

        const hIdsInitial = getHarvestableIDs(farmDoc);
        const pIdsInitial = getPlantableIDs(farmDoc);
        const isReady = (hIdsInitial.length > 0 || pIdsInitial.length > 0);

        const btnStyle = 'padding: 3px 8px; font-size: 11px; white-space: nowrap; color: white; border: none; border-radius: 4px; cursor: pointer; transition: 0.3s;';

        const nBtn = document.createElement('button');
        nBtn.innerHTML = '♻️ เริ่มฤดูกาลใหม่';
        nBtn.style = btnStyle + `background: ${isReady ? '#28a745' : '#dc3545'};`;

        const runNewSeason = async (triggerBtn) => {
            const currentFarmDoc = isFarmPage ? document : await fetchFarmDoc();
            const moneyEl = currentFarmDoc.getElementById('money');
            let currentZen = 0;
            if (moneyEl) {
                currentZen = parseInt(moneyEl.innerText.replace(/,/g, ''), 10) || 0;
            }
            if (currentZen < 25000) {
                timeSpan.innerHTML = '⚠️ Zen ไม่พอซื้อเมล็ดพันธุ์';
                timeSpan.style.color = '#ffeb3b';
                if (triggerBtn && triggerBtn.innerHTML.includes('ออโต้')) {
                    localStorage.setItem('tdFarmAuto', 'false');
                    triggerBtn.style.background = '#6c757d';
                    triggerBtn.innerHTML = '🤖 ออโต้ : ปิด';
                } else if (triggerBtn) {
                    triggerBtn.disabled = false;
                    triggerBtn.innerHTML = '♻️ เริ่มฤดูกาลใหม่';
                }
                alert('❌ Zen ไม่พอซื้อเมล็ดพันธุ์ (ต้องมีอย่างน้อย 25,000 Zen)');
            }

            const hIds = getHarvestableIDs(currentFarmDoc);
            const pIds = getPlantableIDs(currentFarmDoc);

            if (hIds.length === 0 && pIds.length === 0) {
                return alert('❌ ไม่พบแปลงที่พร้อมเก็บหรือปลูก');
            }

            if (triggerBtn) {
                triggerBtn.disabled = true;
                triggerBtn.innerHTML = '⏳ กำลังทำงาน...';
            }

            if (hIds.length > 0) {
                await processActions(hIds, 'store');
                await new Promise(r => setTimeout(r, 500));
            }

            const allToPlant = [...new Set([...pIds, ...hIds])];
            if (allToPlant.length > 0) {
                await processActions(allToPlant, 'seed');
            }

            startReloadCountdown();
        };
        nBtn.onclick = () => runNewSeason(nBtn);

        const isAuto = localStorage.getItem('tdFarmAuto') === 'true';
        const autoBtn = document.createElement('button');
        autoBtn.style = btnStyle + `background: ${isAuto ? '#ff9800' : '#6c757d'}; color: #fff;`;
        autoBtn.innerHTML = isAuto ? '🤖 ออโต้ : เปิด' : '🤖 ออโต้ : ปิด';

        autoBtn.onclick = () => {
            localStorage.setItem('tdFarmAuto', !isAuto);
            initBot();
        };

        const timeSpan = document.createElement('span');
        timeSpan.style = 'color: #eee; font-weight: bold; margin-left: 5px;';

        content.appendChild(nBtn);
        content.appendChild(autoBtn);
        content.appendChild(timeSpan);

        const menuToggler = document.querySelector('button.navbar-toggler[data-toggle="minimize"]');

        if (menuToggler) {
            menuToggler.parentNode.insertBefore(container, menuToggler.nextSibling);
        } else {
            container.style.position = 'fixed';
            container.style.bottom = '10px';
            container.style.right = '10px';
            container.style.background = 'rgba(0,0,0,0.7)';
            container.style.padding = '8px';
            container.style.borderRadius = '8px';
            container.style.zIndex = '9999';
            document.body.appendChild(container);
        }

        let initialSeconds = -1;
        const startTime = Date.now();

        const timeElements = farmDoc.querySelectorAll('.f10');
        timeElements.forEach(el => {
            const text = el.innerText.trim();
            const match = text.match(/\((\d+)\s*วัน\)\s*(\d+):(\d+):(\d+)/);
            if (match) {
                const totalSeconds = (parseInt(match[1]) * 86400) + (parseInt(match[2]) * 3600) + (parseInt(match[3]) * 60) + parseInt(match[4]);
                if (totalSeconds > initialSeconds) initialSeconds = totalSeconds;
            }
        });

        const TARGET_SECONDS = 6 * 3600;
        let countdownToAuto = 10;

        timerInterval = setInterval(() => {
            const realElapsedSinceStart = Math.floor((Date.now() - startTime) / 1000);
            const currentElapsed = initialSeconds + realElapsedSinceStart;

            if (isReady) {
                timeSpan.innerHTML = 'ผลผลิตพร้อมเก็บแล้ว';
            } else if (initialSeconds >= 0) {
                const remaining = Math.max(0, TARGET_SECONDS - currentElapsed);
                if (remaining <= 0) {
                    if (!isReady) {
                        clearInterval(timerInterval);
                        let delay = 20;
                        const delayInterval = setInterval(() => {
                            timeSpan.innerHTML = `กำลังเช็คแปลงในอีก ${delay}`;
                            if (delay <= 0) {
                                clearInterval(delayInterval);
                                initBot();
                            }
                            delay--;
                        }, 1000);
                    }
                } else {
                    const h = Math.floor(remaining / 3600).toString().padStart(2, '0');
                    const m = Math.floor((remaining % 3600) / 60).toString().padStart(2, '0');
                    const s = (remaining % 60).toString().padStart(2, '0');
                    timeSpan.innerHTML = `เวลาคงเหลือ:${h}:${m}:${s}`;
                }
            } else {
                timeSpan.innerHTML = 'สถานะ: ไม่พร้อมใช้งาน';
            }

            if (isAuto && isReady) {
                autoBtn.innerHTML = `🤖 ออโต้ : เปิด (จะทำงานในอีก ${countdownToAuto})`;
                autoBtn.style.background = '#28a745';
                autoBtn.style.color = '#fff';
                countdownToAuto--;
                if (countdownToAuto < 0) {
                    clearInterval(timerInterval);
                    runNewSeason(autoBtn);
                }
            }
        }, 1000);
        setTimeout(() => {
            if (isAuto && isReady) return;

            if (timerInterval) clearInterval(timerInterval);

            let delay = 20;
            const checkInterval = setInterval(() => {
                timeSpan.innerHTML = `กำลังเช็คแปลงในอีก ${delay}`;
                if (delay <= 0) {
                    clearInterval(checkInterval);
                    console.log('🔄 ครบ 30 นาที: กำลังอัปเดตข้อมูลจากเซิร์ฟเวอร์...');
                    initBot();
                }
                delay--;
            }, 1000);
        }, 30 * 60 * 1000);
    }

    async function requestWakeLock() {
        try {
            if ('wakeLock' in navigator && document.visibilityState === 'visible') {
                wakeLock = await navigator.wakeLock.request('screen');
                console.log('✅ ระบบป้องกันการสลีปทำงานแล้ว');
                wakeLock.addEventListener('release', () => {
                    console.log('⚠️ Wake Lock หลุด (อาจเพราะพับจอหรือสลับแท็บ)');
                    wakeLock = null;
                });
            }
        } catch (err) { }
    }

    initBot();
    requestWakeLock();
    document.addEventListener('visibilitychange', () => {
        if (wakeLock === null && document.visibilityState === 'visible') requestWakeLock();
    });
})();
